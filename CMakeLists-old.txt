cmake_minimum_required(VERSION 3.28)
set(CMAKE_CUDA_ARCHITECTURES "native")
project(SubgraphMatching LANGUAGES CXX CUDA)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

set(CUDA_NVCC_FLAGS "${CUDA_NVCC_FLAGS};-std=c++17")

include_directories(.)

add_executable(SubgraphMatching
    main.cu  
    candidate.cu
    join_bfs.cu
    graph.cu
)

set_target_properties(SubgraphMatching PROPERTIES CUDA_SEPARABLE_COMPILATION ON)
